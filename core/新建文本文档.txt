    def _match_wildcard(self, item: Path, config: Dict, context: ProcessingContext) -> bool:
        for pattern, action in config.items():
            if isinstance(pattern, str):
                if (pattern.startswith("**/") and 
                    (fnmatch.fnmatch(str(item), f"**/{pattern[3:]}") or fnmatch.fnmatch(item.name, pattern[3:]))):
                    if isinstance(action, str) and action in self.processors:
                        self._exec(action, item, context)
                        return True
                elif pattern.startswith("*.") and fnmatch.fnmatch(item.name, pattern):
                    if isinstance(action, str) and action in self.processors:
                        self._exec(action, item, context)
                        return True
        return False

    def _process_file(self, file: Path, config: Dict, context: ProcessingContext):
        if self._match_wildcard(file, config, context):
            return
        if file.name in config:
            action = config[file.name]
            if isinstance(action, str) and action in self.processors:
                self._exec(action, file, context)

    def _exec(self, actions: Union[str, List[str]], path: Path, context: ProcessingContext):
        if isinstance(actions, str):
            actions = [actions]
        
        for action_name in actions:
            if action_name not in self.processors:
                print(f"âŒ æœªçŸ¥å¤„ç†å™¨: {action_name}")
                continue
            
            func = self.processors[action_name]
            try:
                print(f"  â¡ï¸  [{action_name}] {path}")
                
                # ğŸ‘‰ æ”¯æŒç”Ÿæˆå™¨ï¼ˆyield å¤šæ¡ï¼‰
                result = func(path, context)    ##è¿è¡Œå¤„ç†å‡½æ•°
                if isinstance(result, Generator):
                    for item in result:
                        if isinstance(item, dict):
                            context.add_result(item)
                elif isinstance(result, dict):
                    context.add_result(result)
                # å…¶ä»–ç±»å‹å¿½ç•¥
                
                ## resultä¸ºFalseæ—¶
                if isinstance(result, dict):
                    return result.get("continue", True)
                return True
            except Exception as e:
                # å³ä½¿å¤±è´¥ä¹Ÿè®°å½•ä¸€è¡Œ
                context.add_result({
                    "file": str(path),
                    "processor": action_name,
                    "status": "error",
                    "error": str(e),
                    "timestamp": self._now()
                })                
                
                ##å‡ºé”™æ—¶ä¸å†ç»§ç»­
                return False
           
            return True
